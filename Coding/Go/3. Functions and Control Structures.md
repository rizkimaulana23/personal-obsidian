## nil
`nil` is the default value for different types. For example, the default value of Error is nil.
## Error
```go
var err error // default value is nil
```

```go
package main

  

import (

    "errors"

    "fmt"

)

  

func main() {

    var printValue string = "Hello World"

    printMe(printValue)

  

    var numerator = 11

    var denomerator = 1

    var result, remainder, err = intDivision(numerator, denomerator)

    if err != nil {

        fmt.Println(err.Error())

    } else if remainder == 0 {

        fmt.Printf("The result of the integer division is %v", result)

    } else {

        fmt.Printf("The result of the integer division is %v with remainder S%v", result, remainder)

    }

  

    switch {

    case err != nil:

        fmt.Println(err.Error())

    }

  

    switch err {

    case nil:

        fmt.Println("No Error what so ever")

    default:

        fmt.Println(err.Error())

    }

}

  

func printMe(printValue string) {

    fmt.Println(printValue)

}

  

func intDivision(numerator int, denomerator int) (int, int, error) {

    var err error

    if denomerator == 0 {

        err = errors.New("cannot divide by zero")

        return 0, 0, err

    }

    var result int = numerator / denomerator

    var remainder = numerator % denomerator

    return result, remainder, err

}
```
